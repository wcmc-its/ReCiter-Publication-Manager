    ```typescript
    /**
     * Scope resolution for Curator_Scoped role.
     * Centralized pure function -- used by Edge middleware (via JWT), API routes, and React components.
     * AND across dimensions (person type AND org unit), OR within each dimension.
     */

    export interface ScopeData {
      personTypes: string[] | null;  // null = no restriction on person type axis
      orgUnits: string[] | null;     // null = no restriction on org unit axis
    }

    /**
     * Check if a person matches the curator's scope.
     *
     * Rules:
     * - null dimension = no restriction on that axis (matches everything)
     * - AND across dimensions: person must match BOTH person type AND org unit restrictions (if set)
     * - OR within each dimension: person matches if ANY of their types matches ANY of the scope's types
     * - Person with no person types + scope requiring person types = no match
     * - Person with null org unit + scope requiring org units = no match
     *
     * @param scope - The curator's scope data from JWT
     * @param personOrgUnit - The person's primaryOrganizationalUnit (can be null)
     * @param personPersonTypes - Array of the person's person types from person_person_type table
     * @returns boolean - true if person is within the curator's scope
     */
    export function isPersonInScope(
      scope: ScopeData,
      personOrgUnit: string | null,
      personPersonTypes: string[]
    ): boolean {
      // No scope restrictions = matches everything (safety net)
      if (!scope.personTypes && !scope.orgUnits) return true;

      // Check person type dimension (if restricted)
      if (scope.personTypes && scope.personTypes.length > 0) {
        const hasMatchingType = personPersonTypes.some(pt =>
          scope.personTypes!.includes(pt)
        );
        if (!hasMatchingType) return false;
      }

      // Check org unit dimension (if restricted)
      if (scope.orgUnits && scope.orgUnits.length > 0) {
        if (!personOrgUnit || !scope.orgUnits.includes(personOrgUnit)) {
          return false;
        }
      }

      return true;
    }
    ```
