    /**
     * Get a person's org unit and person types for scope checking.
     * Lightweight query used by API-level scope enforcement.

    export const getPersonWithTypes = async (uid: string) => {
      try {
        const person = await models.Person.findOne({
          where: { personIdentifier: uid },
          attributes: ["personIdentifier", "primaryOrganizationalUnit"],
        });
        if (!person) return null;

        const personTypes = await models.PersonPersonType.findAll({
          where: { personIdentifier: uid },
          attributes: ["personType"],

          primaryOrganizationalUnit: person.primaryOrganizationalUnit,
          personTypes: personTypes.map((pt: any) => pt.personType),
        };

    const { getCapabilities } = require('../../../../../../src/utils/constants');
    const { isPersonInScope } = require('../../../../../../src/utils/scopeResolver');
    const { getPersonWithTypes } = require('../../../../../../controllers/db/person.controller');

    const token = await getToken({ req, secret: reciterConfig.tokenSecret });
    if (token && token.userRoles) {
      const roles = JSON.parse(token.userRoles);
      const caps = getCapabilities(roles);

      if (caps.canCurate.scoped && !caps.canCurate.all) {
        const scopeData = token.scopeData ? JSON.parse(token.scopeData) : null;
        if (scopeData) {
          const personData = await getPersonWithTypes(uid as string);
          if (!personData) {
            console.log('[AUTH] DENY: Person not found for scope check:', uid);

          const inScope = isPersonInScope(scopeData, personData.primaryOrganizationalUnit, personData.personTypes);
          if (!inScope) {
            console.log('[AUTH] DENY: Scoped curator', token.username, 'tried to save feedback for', uid, '-- not in scope');
            return res.status(403).json({ error: 'Person not in curation scope' });
